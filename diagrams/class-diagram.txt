enum UserRole {
  ADMIN
  USER
}

enum UserStatus {
  ACTIVE
  DISABLED
}

enum PermissionType {
  READ
  EDIT
}

enum TrashStatus {
  ACTIVE
  TRASHED
}

class User {
  -id: UUID
  -username: String
  -email: String
  -passwordHash: String
  -role: UserRole
  -status: UserStatus
  -storageQuotaBytes: Long
  -storageUsedBytes: Long
  -createdAt: DateTime
  -updatedAt: DateTime
  +changePassword(newPassword: String): void
  +getStorageUsage(): Long
  +isAdmin(): Boolean
}

class AccessToken {
  -tokenValue: String
  -expiresAt: DateTime
  +isExpired(): Boolean
}

class RefreshToken {
  -id: UUID
  -tokenHash: String
  -expiresAt: DateTime
  -revoked: Boolean
  -createdAt: DateTime
  +isExpired(): Boolean
  +revoke(): void
}

class SystemConfig {
  -isInitialized: Boolean
  -storageRootPath: String
  -totalStorageLimitBytes: Long
  -createdAt: DateTime
  -updatedAt: DateTime
  +initialize(): void
  +updateStorageRoot(path: String): void
  +updateStorageLimit(limit: Long): void
}

abstract class StorageEntry {
  -id: UUID
  -name: String
  -path: String
  -isStarred: Boolean
  -trashStatus: TrashStatus
  -createdAt: DateTime
  -updatedAt: DateTime
  +rename(newName: String): void
  +move(targetPath: String): void
  +moveToTrash(): void
  +restore(): void
  +setStarred(starred: Boolean): void
}

class Folder {
  +createSubFolder(name: String): Folder
  +listEntries(): List<StorageEntry>
}

class File {
  -mimeType: String
  -extension: String
  -sizeBytes: Long
  -checksum: String
  -physicalPath: String
  +download(): Binary
  +preview(): Binary
}

class SharePermission {
  -id: UUID
  -permissionType: PermissionType
  -createdAt: DateTime
  +updatePermission(permission: PermissionType): void
}

class AuthService {
  +login(username: String, password: String): AccessToken
  +refresh(refreshToken: String): AccessToken
  +validateAccessToken(token: String): User
  +revokeRefreshToken(token: String): void
}

class UserService {
  +createUser(username: String, email: String, password: String): User
  +updateUser(userId: UUID): User
  +disableUser(userId: UUID): void
  +listUsers(): List<User>
  +findUserById(userId: UUID): User
}

class StorageService {
  +listEntries(userId: UUID, path: String): List<StorageEntry>
  +createFolder(userId: UUID, parentPath: String, name: String): Folder
  +uploadFile(userId: UUID, parentPath: String, fileName: String): File
  +renameEntry(path: String, newName: String): StorageEntry
  +moveEntry(sourcePath: String, targetPath: String): StorageEntry
  +deleteEntry(path: String): void
  +restoreEntry(path: String): void
  +setStarred(path: String, starred: Boolean): StorageEntry
}

class ShareService {
  +shareEntry(ownerId: UUID, targetUserId: UUID, path: String, permission: PermissionType): SharePermission
  +updateSharePermission(shareId: UUID, permission: PermissionType): SharePermission
  +removeShare(shareId: UUID): void
  +listSharedEntries(userId: UUID): List<StorageEntry>
}

class AdminService {
  +initializeSystem(adminUser: User, config: SystemConfig): void
  +getSystemConfig(): SystemConfig
  +updateSystemConfig(config: SystemConfig): SystemConfig
  +getAllUsers(): List<User>
}
