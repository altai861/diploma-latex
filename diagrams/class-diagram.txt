package "Boundary" {
  class SetupBoundary <<boundary>> {
    +checkSetupStatus(): SetupStatus
    +initializeSystem(adminData, systemData)
  }

  class AuthBoundary <<boundary>> {
    +login(username, password)
    +logout(accessToken)
    +getCurrentUser(accessToken)
    +updateProfileImage(accessToken, imageFile)
  }

  class StorageBoundary <<boundary>> {
    +listResources(accessToken, path)
    +createFolder(accessToken, parentPath, folderName)
    +uploadFile(accessToken, parentPath, file)
    +renameResource(accessToken, resourceId, newName)
    +moveResource(accessToken, resourceId, targetFolderId)
    +deleteResource(accessToken, resourceId)
    +restoreResource(accessToken, resourceId)
    +downloadFile(accessToken, fileId)
    +downloadBatch(accessToken, resourceIds)
    +searchResources(accessToken, keyword)
    +setStarred(accessToken, resourceId, starred)
  }

  class ShareBoundary <<boundary>> {
    +listSharedResources(accessToken)
    +listPermissions(accessToken, resourceId)
    +grantPermission(accessToken, resourceId, targetUserId, accessLevel)
    +removePermission(accessToken, resourceId, targetUserId)
    +searchShareableUsers(accessToken, keyword)
  }

  class AdminBoundary <<boundary>> {
    +listUsers(accessToken)
    +createUser(accessToken, userData)
    +updateUser(accessToken, userId, userData)
    +deleteUser(accessToken, userId)
  }
}

package "Control" {
  class SetupController <<control>> {
    +isInitialized(): Boolean
    +initializeSystem(adminData, systemData)
  }

  class AuthenticationController <<control>> {
    +login(username, password): Session
    +authenticate(accessToken): User
    +logout(accessToken)
    +updateProfileImage(userId, imageFile)
  }

  class StorageController <<control>> {
    +listResources(userId, path)
    +getMetadata(userId, resourceId)
    +createFolder(userId, parentPath, folderName): Folder
    +uploadFile(userId, parentPath, file): File
    +renameResource(userId, resourceId, newName): StorageEntry
    +moveResource(userId, resourceId, targetFolderId): StorageEntry
    +deleteResource(userId, resourceId)
    +restoreResource(userId, resourceId)
    +downloadFile(userId, fileId): File
    +downloadBatch(userId, resourceIds)
    +searchResources(userId, keyword)
    +setStarred(userId, resourceId, starred): StorageEntry
  }

  class SharingController <<control>> {
    +listSharedResources(userId)
    +listPermissions(ownerUserId, resourceId)
    +grantPermission(ownerUserId, resourceId, targetUserId, accessLevel): SharePermission
    +removePermission(ownerUserId, resourceId, targetUserId)
    +searchShareableUsers(userId, keyword)
  }

  class UserManagementController <<control>> {
    +listUsers(adminUserId)
    +createUser(adminUserId, userData): User
    +updateUser(adminUserId, userId, userData): User
    +deleteUser(adminUserId, userId)
  }
}

package "Entity" {
  enum UserRole {
    ADMIN
    USER
  }

  enum AccessLevel {
    OWNER
    EDITOR
    VIEWER
  }

  abstract class StorageEntry <<entity>> {
    +id: Long
    +name: String
    +path: String
    +isDeleted: Boolean
    +isStarred: Boolean
    +createdAt: DateTime
    +updatedAt: DateTime
  }

  class User <<entity>> {
    +id: Long
    +username: String
    +email: String
    +fullName: String
    +passwordHash: String
    +role: UserRole
    +status: String
    +storageQuotaBytes: Long
    +storageUsedBytes: Long
  }

  class Session <<entity>> {
    +id: Long
    +tokenHash: String
    +expiresAt: DateTime
    +revokedAt: DateTime
  }

  class SystemConfig <<entity>> {
    +isInitialized: Boolean
    +storageRootPath: String
    +totalStorageLimitBytes: Long
  }

  class Folder <<entity>> {
    +parentFolderId: Long
  }

  class File <<entity>> {
    +sizeBytes: Long
    +mimeType: String
    +extension: String
    +storagePath: String
    +checksum: String
  }

  class SharePermission <<entity>> {
    +id: Long
    +accessLevel: AccessLevel
    +createdAt: DateTime
  }
}
